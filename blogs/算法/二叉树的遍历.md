#### 二叉树的遍历之递归写法

1. 前序遍历

```js
var inorderTraversal = function(root, arr = []) {
	if (!root) return arr;
	arr.push(root.val);
	inorderTraversal(root.left, arr);
	inorderTraversal(root.right, arr);
	return arr;
};
```

2. 中序遍历

```js
var inorderTraversal = function(root, arr = []) {
	if (!root) return arr;
	inorderTraversal(root.left, arr);
	arr.push(root.val);
	inorderTraversal(root.right, arr);
	return arr;
};
```

3. 后序遍历

```js
var inorderTraversal = function(root, arr = []) {
	if (!root) return arr;
	inorderTraversal(root.left, arr);
	inorderTraversal(root.right, arr);
	arr.push(root.val);
	return arr;
};
```

### 二叉树的遍历之迭代写法

1. 前序遍历

```js
var inorderTraversal = function(root, stack = [], arr = []) {
	while (stack.length || root) {
		while (root) {
			stack.push(root);
			arr.push(root.val);
			root = root.left;
		}
		root = stack.pop();
		root = root.right;
	}
	return arr;
};
```

2. 中序遍历

```js
var inorderTraversal = function(root, stack = [], arr = []) {
	while (stack.length || root) {
		while (root) {
			stack.push(root);
			root = root.left;
		}
		root = stack.pop();
		arr.push(root.val);
		root = root.right;
	}
	return arr;
};
```

3. 后序遍历

```js
var inorderTraversal = function(root, stack = [], arr = []) {
	while (stack.length || root) {
		while (root) {
			stack.push(root);
			arr.unshift(root.val);
			root = root.right;
		}
		root = stack.pop();
		root = root.left;
	}
	return arr;
};
```

#### 解析迭代过程

以上迭代 1 和 2 的过程解析，入栈出栈过程如下
1 进->1 出->2 进->3 进->3 出->2 出

所以进栈的顺序为 1->2->3 刚好符合前序遍历顺序
出栈顺序为 1->3->2 刚好符合中序遍历的顺序
