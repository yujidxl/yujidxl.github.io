#### 二叉树的遍历之递归写法
1. 前序遍历
``` js
var inorderTraversal = function(root, arr = []) {
    if (!root) return arr;
    arr.push(root.val);
    inorderTraversal(root.left, arr);
    inorderTraversal(root.right, arr);
    return arr;
};
```
2. 中序遍历
```js
var inorderTraversal = function(root, arr = []) {
    if (!root) return arr;
    inorderTraversal(root.left, arr);
    arr.push(root.val);
    inorderTraversal(root.right, arr);
    return arr;
};
```
3. 后序遍历
``` js
var inorderTraversal = function(root, arr = []) {
    if (!root) return arr;
    inorderTraversal(root.left, arr);
    inorderTraversal(root.right, arr);
    arr.push(root.val);
    return arr;
};
```
### 二叉树的遍历之迭代写法
1. 前序遍历
```js
var inorderTraversal = function(root, stack = [], arr = []) {
    while(stack.length || root) {
        while(root) {
            stack.push(root);
            arr.push(root.val);
            root = root.left;
        }
        root = stack.pop();
        root = root.right;
    }
    return arr;
};
```
2. 中序遍历
```js
var inorderTraversal = function(root, stack = [], arr = []) {
    while(stack.length || root) {
        while(root) {
            stack.push(root);
            arr.push(root.val);
            root = root.left;
        }
        root = stack.pop();
        root = root.right;
    }
    return arr;
};
```
3. 后序遍历
```js
var inorderTraversal = function(root, stack = [], arr = []) {
    while(stack.length || root) {
        while(root) {
            stack.push(root);
            arr.unshift(root.val);
            root = root.right;
        }
        root = stack.pop();
        root = root.left;
    }
    return arr;
};
```

 #### 解析迭代过程
 以上迭代1和2的过程解析，入栈出栈过程如下
 1进->1出->2进->3进->3出->2出

 所以进栈的顺序为1->2->3刚好符合前序遍历顺序
 出栈顺序为1->3->2刚好符合中序遍历的顺序
